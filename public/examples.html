Same drop-in

Each site
uses one snippet.

Five surfaces, five tiers, one piece of middleware. Per-route config picks which paths to gate & how strict.

Build your own → 
// express
const stile = require('@stile/express');

app.use(stile({
  secret: process.env.STILE_SECRET,
  routes: [
    { path: '/api/article/*',     tier: 'easy'   },
    { path: '/openapi.json',      tier: 'medium' },
    { path: '/feed/products.json',tier: 'medium' },
    { path: '/listings.jsonl',    tier: 'strong' },
    { path: '/obs/*',             tier: 'strong' },
  ],
  honeypot: true,
}));
