Composition №04 — Plug & play

One file. Zero dependencies.

node http 
const http = require('http');
const stile = require('stile');

const g = stile({
  secret: env.STILE_SECRET,
  protect: ['/agents'],
});
http.createServer(g.wrap(app))
    .listen(3000);
express 
const stile = require(
  'stile/adapters/express'
);

app.use(stile({
  secret: env.STILE_SECRET,
  protect: ['/agents'],
  tier: 'easy',
}));
hono 
import stile from
  'stile/adapters/hono';

app.use('*', stile({
  secret: Bun.env.STILE_SECRET,
  protect: ['/agents'],
}));
next · middleware.ts 
import stile from
  'stile/adapters/next';

export const config = {
  matcher: ['/agents/:path*'],
  runtime: 'nodejs',
};
export default stile({
  protect: ['/agents'],
});
